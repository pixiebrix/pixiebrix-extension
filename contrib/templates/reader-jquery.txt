apiVersion: v1
kind: reader
metadata:
  id: apartments.com/property-reader
  version: 0.0.1
  name: Apartments.com Property Info
  description: Read property information from Apartments.com
  author: pixiebrix
definition:
  reader:
    type: jquery
    selectors:
      propertyName: "#propertyHeader .propertyName"
      propertyAddress: "#propertyHeader .propertyAddress h2"
      neighborhood: "#propertyHeader .neighborhoodAddress > a.neighborhood"
      propertySize:
        selector: ".specList:has(h3:contains('Property Information')) ul li:contains('Units')"
        contents: text
      propertyYearBuilt:
        selector: ".specList:has(h3:contains('Property Information')) ul li:contains('Built')"
        contents: text
      propertyCompany:
        selector: "#propertyHeader img.logo"
        attr: alt
      propertyPhone: "#contactLead #contactLeadForm .contactPhone"
      availability:
        selector: tr.rentalGridRow
        multi: true
        find:
          maxrent:
            data: maxrent
          beds:
            data: beds
          model:
            data: model
          baths:
            data: baths
          unit:
            data: unit
      website:
        selector: "#contactLead .linkWrapper a[title='View Property Website']"
        attr: href
      agent:
        selector: "#contactSection"
        find:
          name: .agentFullName
          phone: .phoneNumber
  isAvailable:
    matchPatterns: "https://www.apartments.com/*"
    selectors: "#profileHeaderWrapper"
outputSchema:
  type: object
  properties:
    propertyName:
      type: string
    propertyAddress:
      type: string
    neighborhood:
      type: string
    propertySize:
      type: string
    propertyYearBuilt:
      type: string
    propertyCompany:
      type: string
    propertyPhone:
      type: string
    availability:
      type: array
      items:
        type: object
        properties:
          maxrent:
            type: number
          beds:
            type: number
          model:
            type: number
          bath:
            type: number
          unit:
            type: number
    website:
      type: string
      format: uri
    agent:
      type: object
      properties:
        name:
          type: string
        phone:
          type: string
